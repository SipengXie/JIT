32/// The Cranelift-based EVM bytecode compiler backend.
33#[allow(missing_debug_implementations)]
34#[must_use]
35pub struct EvmCraneliftBackend {
36    /// The function builder context, which is reused across multiple FunctionBuilder instances.
37    builder_context: FunctionBuilderContext,
38
39    /// The main Cranelift context, which holds the state for codegen. Cranelift
40    /// separates this from `Module` to allow for parallel compilation, with a
41    /// context per thread, though this isn't in the simple demo here.
42    ctx: codegen::Context,
43
44    /// The module, with the jit backend, which manages the JIT'd functions.
45    module: ModuleWrapper,
46
47    symbols: Symbols,
48
49    opt_level: OptimizationLevel,
50    comments: CommentWriter,
51    functions: Vec<FuncId>,
52}

54#[allow(clippy::new_without_default)]
55impl EvmCraneliftBackend {
56    /// Returns `Ok(())` if the current architecture is supported, or `Err(())` if the host machine
57    /// is not supported in the current configuration.
58    pub fn is_supported() -> Result<(), &'static str> {
59        cranelift_native::builder().map(drop)
60    }
61
62    /// Creates a new instance of the JIT compiler.
63    ///
64    /// # Panics
65    ///
66    /// Panics if the current architecture is not supported. See
67    /// [`is_supported`](Self::is_supported).
68    #[track_caller]
69    pub fn new(aot: bool, opt_level: OptimizationLevel) -> Self {
70        let symbols = Symbols::new();
71        let module = ModuleWrapper::new(aot, opt_level, &symbols).unwrap();
72        Self {
73            builder_context: FunctionBuilderContext::new(),
74            ctx: module.get().make_context(),
75            module,
76            symbols,
77            opt_level,
78            comments: CommentWriter::new(),
79            functions: Vec::new(),
80        }
81    }
82
83    fn finish_module(&mut self) -> Result<Option<ObjectModule>> {
84        let aot = match self.module {
85            ModuleWrapper::Jit(_) => {
86                // TODO: Can `free_memory` take `&mut self` pls?
87                let new = ModuleWrapper::new_jit(self.opt_level, self.symbols.clone())?;
88                let ModuleWrapper::Jit(old) = std::mem::replace(&mut self.module, new) else {
89                    unreachable!()
90                };
91                unsafe { old.free_memory() };
92                None
93            }
94            ModuleWrapper::Aot(_) => {
95                let new = ModuleWrapper::new_aot(self.opt_level)?;
96                let ModuleWrapper::Aot(old) = std::mem::replace(&mut self.module, new) else {
97                    unreachable!()
98                };
99                Some(old)
100            }
101        };
102        self.module.get().clear_context(&mut self.ctx);
103        Ok(aot)
104    }
105}

192    fn build_function(
193        &mut self,
194        name: &str,
195        ret: Option<Self::Type>,
196        params: &[Self::Type],
197        param_names: &[&str],
198        linkage: revmc_backend::Linkage,
199    ) -> Result<(Self::Builder<'_>, FuncId)> {
200        self.ctx.func.clear();
201        if let Some(ret) = ret {
202            self.ctx.func.signature.returns.push(AbiParam::new(ret));
203        }
204        for param in params {
205            self.ctx.func.signature.params.push(AbiParam::new(*param));
206        }
207        let _ = param_names;
208        let ptr_type = self.type_ptr();
209        let id = self.module.get_mut().declare_function(

215        let bcx = FunctionBuilder::new(&mut self.ctx.func, &mut self.builder_context);
216        let mut builder = EvmCraneliftBuilder {
217            module: &mut self.module,
218            comments: &mut self.comments,

220            ptr_type,
221            symbols: self.symbols.clone(),

232    fn optimize_module(&mut self) -> Result<()> {
233        // Define the function to jit. This finishes compilation, although
234        // there may be outstanding relocations to perform. Currently, jit
235        // cannot finish relocations until all functions to be called are
236        // defined. For this toy demo for now, we'll just finalize the
237        // function below.
238        for &id in &self.functions {
239            self.module.get_mut().define_function(id, &mut self.ctx)?;
240        }
241        self.functions.clear();
242
243        // Now that compilation is finished, we can clear out the context state.
244        self.module.get().clear_context(&mut self.ctx);
245
246        // Finalize the functions which we just defined, which resolves any outstanding relocations
247        // (patching in addresses, now that they're available).
248        self.module.finalize_definitions()?;
249
250        self.comments.clear();
251
252        Ok(())
253    }

255    fn write_object<W: std::io::Write>(&mut self, w: W) -> Result<()> {
256        let module =
257            self.finish_module()?.ok_or_else(|| eyre!("cannot write object in JIT mode"))?;
258        let product = module.finish();
259        product.object.write_stream(w).map_err(|e| eyre!("{e}"))?;
260        Ok(())
261    }

542    fn phi(&mut self, ty: Self::Type, incoming: &[(Self::Value, Self::BasicBlock)]) -> Self::Value {
543        let current = self.current_block().unwrap();
544        let param = self.bcx.append_block_param(current, ty);
545        for &(value, block) in incoming {
546            self.bcx.switch_to_block(block);
547            let last_inst = self.bcx.func.layout.last_inst(block).unwrap();
548            let src = self.bcx.ins().jump(current, &[value]);
549            self.bcx.func.transplant_inst(last_inst, src);
550        }
551        self.bcx.switch_to_block(current);
552        param
553    }

564    fn lazy_select(
565        &mut self,
566        cond: Self::Value,
567        ty: Self::Type,
568        then_value: impl FnOnce(&mut Self) -> Self::Value,
569        else_value: impl FnOnce(&mut Self) -> Self::Value,
570    ) -> Self::Value {
571        let then_block = if let Some(current) = self.current_block() {
572            self.create_block_after(current, "then")
573        } else {
574            self.create_block("then")
575        };
576        let else_block = self.create_block_after(then_block, "else");
577        let done_block = self.create_block_after(else_block, "contd");
578        let done_value = self.bcx.append_block_param(done_block, ty);
579
580        self.brif(cond, then_block, else_block);
581
582        self.seal_block(then_block);
583        self.switch_to_block(then_block);
584        let then_value = then_value(self);
585        self.bcx.ins().jump(done_block, &[then_value]);
586
587        self.seal_block(else_block);
588        self.switch_to_block(else_block);
589        let else_value = else_value(self);
590        self.bcx.ins().jump(done_block, &[else_value]);
591
592        self.seal_block(done_block);
593        self.switch_to_block(done_block);
594        done_value
595    }

751    fn get_or_build_function(
752        &mut self,
753        name: &str,
754        params: &[Self::Type],
755        ret: Option<Self::Type>,
756        linkage: revmc_backend::Linkage,
757        build: impl FnOnce(&mut Self),
758    ) -> Self::Function {
759        if let Some(f) = self.get_function(name) {
760            return f;
761        }
762
763        let mut sig = self.module.get().make_signature();
764        if let Some(ret) = ret {
765            sig.returns.push(AbiParam::new(ret));
766        }
767        for param in params {
768            sig.params.push(AbiParam::new(*param));
769        }
770
771        let id =
772            self.module.get_mut().declare_function(name, convert_linkage(linkage), &sig).unwrap();
773
774        let mut func = Function::new();
775        func.signature = sig;
776        let mut builder_ctx = FunctionBuilderContext::new();
777        let new_bcx = FunctionBuilder::new(&mut func, &mut builder_ctx);
778        // TODO: SAFETY: Not really safe, lifetime extension.
779        let new_bcx =
780            unsafe { std::mem::transmute::<FunctionBuilder<'_>, FunctionBuilder<'a>>(new_bcx) };
781        let old_bcx = std::mem::replace(&mut self.bcx, new_bcx);
782
783        let f = self.module.get_mut().declare_func_in_func(id, self.bcx.func);
784
785        let entry = self.bcx.create_block();
786        self.bcx.append_block_params_for_function_params(entry);
787        build(self);
788
789        self.bcx = old_bcx;
790
791        f
792    }

794    fn get_function(&mut self, name: &str) -> Option<Self::Function> {
795        self.module
796            .get()
797            .get_name(name)
798            .and_then(|id| match id {
799                FuncOrDataId::Func(f) => Some(f),
800                FuncOrDataId::Data(_) => None,
801            })
802            .map(|id| self.module.get_mut().declare_func_in_func(id, self.bcx.func))
803    }

866enum ModuleWrapper {
867    Jit(JITModule),
868    Aot(ObjectModule),
869}
870
871impl ModuleWrapper {
872    fn new(aot: bool, opt_level: OptimizationLevel, symbols: &Symbols) -> Result<Self> {
873        if aot {
874            Self::new_aot(opt_level)
875        } else {
876            Self::new_jit(opt_level, symbols.clone())
877        }
878    }
879
880    fn new_jit(opt_level: OptimizationLevel, symbols: Symbols) -> Result<Self> {
881        let mut builder = JITBuilder::with_flags(
882            &[("opt_level", opt_level_flag(opt_level))],
883            cranelift_module::default_libcall_names(),
884        )?;
885        builder.symbol_lookup_fn(Box::new(move |s| symbols.get(s)));
886        Ok(Self::Jit(JITModule::new(builder)))
887    }
888
889    fn new_aot(opt_level: OptimizationLevel) -> Result<Self> {
890        let mut flag_builder = settings::builder();
891        flag_builder.set("opt_level", opt_level_flag(opt_level))?;
892        let isa_builder = cranelift_native::builder().map_err(|s| eyre!(s))?;
893        let isa = isa_builder.finish(settings::Flags::new(flag_builder))?;
894
895        let builder =
896            ObjectBuilder::new(isa, "jit".to_string(), cranelift_module::default_libcall_names())?;
897        Ok(Self::Aot(ObjectModule::new(builder)))
898    }
899
900    fn is_aot(&self) -> bool {
901        matches!(self, Self::Aot(_))
902    }
903
904    #[inline]
905    fn get(&self) -> &dyn Module {
906        match self {
907            Self::Jit(module) => module,
908            Self::Aot(module) => module,
909        }
910    }
911
912    #[inline]
913    fn get_mut(&mut self) -> &mut dyn Module {
914        match self {
915            Self::Jit(module) => module,
916            Self::Aot(module) => module,
917        }
918    }
919
920    fn finalize_definitions(&mut self) -> Result<(), ModuleError> {
921        match self {
922            Self::Jit(module) => module.finalize_definitions(),
923            Self::Aot(_) => Ok(()),
924        }
925    }
926
927    fn get_finalized_function(&self, id: FuncId) -> Result<*const u8> {
928        match self {
929            Self::Jit(module) => Ok(module.get_finalized_function(id)),
930            Self::Aot(_) => Err(eyre!("cannot get finalized JIT function in AOT mode")),
931        }
932    }
933}

935fn convert_intcc(cond: revmc_backend::IntCC) -> IntCC {
936    match cond {
937        revmc_backend::IntCC::Equal => IntCC::Equal,
938        revmc_backend::IntCC::NotEqual => IntCC::NotEqual,
939        revmc_backend::IntCC::SignedLessThan => IntCC::SignedLessThan,
940        revmc_backend::IntCC::SignedGreaterThanOrEqual => IntCC::SignedGreaterThanOrEqual,
941        revmc_backend::IntCC::SignedGreaterThan => IntCC::SignedGreaterThan,
942        revmc_backend::IntCC::SignedLessThanOrEqual => IntCC::SignedLessThanOrEqual,
943        revmc_backend::IntCC::UnsignedLessThan => IntCC::UnsignedLessThan,
944        revmc_backend::IntCC::UnsignedGreaterThanOrEqual => IntCC::UnsignedGreaterThanOrEqual,
945        revmc_backend::IntCC::UnsignedGreaterThan => IntCC::UnsignedGreaterThan,
946        revmc_backend::IntCC::UnsignedLessThanOrEqual => IntCC::UnsignedLessThanOrEqual,
947    }
948}

950fn convert_linkage(linkage: revmc_backend::Linkage) -> Linkage {
951    match linkage {
952        revmc_backend::Linkage::Import => Linkage::Import,
953        revmc_backend::Linkage::Public => Linkage::Export,
954        revmc_backend::Linkage::Private => Linkage::Local,
955    }
