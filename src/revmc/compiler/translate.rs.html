132impl<'a, B: Backend> FunctionCx<'a, B> {
133    /// Translates an EVM bytecode into a native function.
134    ///
135    /// Example pseudo-code:
136    ///
137    /// ```ignore (pseudo-code)
138    /// // `cfg(may_suspend) = bytecode.may_suspend()`: `true` if it contains a
139    /// // `*CALL*` or `CREATE*` instruction.
140    /// fn evm_bytecode(args: ...) {
141    ///     setup_locals();
142    ///
143    ///     #[cfg(debug_assertions)]
144    ///     if args.<ptr>.is_null() { panic!("...") };
145    ///
146    ///     load_arguments();
147    ///
148    ///     #[cfg(may_suspend)]
149    ///     resume: {
150    ///         goto match ecx.resume_at {
151    ///             0 => inst0,
152    ///             1 => first_call_or_create_inst + 1, // + 1 as in the block after.
153    ///             2 => second_call_or_create_inst + 1,
154    ///             ... => ...,
155    ///             _ => unreachable, // Assumed to be valid.
156    ///         };
157    ///     };
158    ///     
159    ///     op.inst0: { /* ... */ };
160    ///     op.inst1: { /* ... */ };
161    ///     // ...
162    ///     #[cfg(may_suspend)]
163    ///     first_call_or_create_inst: {
164    ///          // ...
165    ///          goto suspend(1);
166    ///     };
167    ///     // ...
168    ///     // There will always be at least one diverging instruction.
169    ///     op.stop: {
170    ///         goto return(InstructionResult::Stop);
171    ///     };
172    ///
173    ///     #[cfg(may_suspend)]
174    ///     suspend(resume_at: u32): {
175    ///         ecx.resume_at = resume_at;
176    ///         goto return(InstructionResult::CallOrCreate);
177    ///     };
178    ///
179    ///     // All paths lead to here.
180    ///     return(ir: InstructionResult): {
181    ///         #[cfg(inspect_stack_length)]
182    ///         *args.stack_len = stack_len;
183    ///         return ir;
184    ///     }
185    /// }
186    /// ```
187    #[allow(rustdoc::invalid_rust_codeblocks)] // Syntax highlighting.
188    pub(super) fn translate(
189        mut bcx: B::Builder<'a>,
190        config: FcxConfig,
191        builtins: &'a mut Builtins<B>,
192        bytecode: &'a Bytecode<'a>,
193    ) -> Result<()> {
194        let entry_block = bcx.current_block().unwrap();
195
196        // Get common types.
197        let ptr_type = bcx.type_ptr();
198        let isize_type = bcx.type_ptr_sized_int();
199        let i8_type = bcx.type_int(8);
200        let i64_type = bcx.type_int(64);
201        let address_type = bcx.type_int(160);
202        let word_type = bcx.type_int(256);
203
204        // Set up entry block.
205        let gas_ptr = bcx.fn_param(0);
206        let gas_remaining = {
207            let offset = bcx.iconst(i64_type, mem::offset_of!(pf::Gas, remaining) as i64);
208            let name = "gas.remaining.addr";
209            Pointer::new_address(i64_type, bcx.gep(i8_type, gas_ptr, &[offset], name))
210        };
211
212        let sp_arg = bcx.fn_param(1);
213        let stack = if config.local_stack {
214            bcx.new_stack_slot(word_type, "stack.addr")
215        } else {
216            Pointer::new_address(word_type, sp_arg)
217        };
218
219        let stack_len_arg = bcx.fn_param(2);
220        // This is initialized later in `post_entry_block`.
221        let stack_len = bcx.new_stack_slot(isize_type, "len.addr");
222
223        let env = bcx.fn_param(3);
224        let contract = bcx.fn_param(4);
225        let ecx = bcx.fn_param(5);
226
227        // Create all instruction entry blocks.
228        let unreachable_block = bcx.create_block("unreachable");
229        let inst_entries: Vec<_> = bytecode
230            .iter_all_insts()
231            .map(|(i, data)| {
232                if data.is_dead_code() {
233                    unreachable_block
234                } else {
235                    bcx.create_block(&bytecode.op_block_name(i, ""))
236                }
237            })
238            .collect();
239        assert!(!inst_entries.is_empty(), "translating empty bytecode");
240
241        let dynamic_jump_table = bcx.create_block("dynamic_jump_table");
242        let suspend_block = bcx.create_block("suspend");
243        let failure_block = bcx.create_block("failure");
244        let return_block = bcx.create_block("return");
245
246        let mut fx = FunctionCx {
247            config,
248
249            ptr_type,
250            isize_type,
251            address_type,
252            word_type,
253            i8_type,
254            stack_len,
255            stack,
256            gas_remaining,
257            env,
258            contract,
259            ecx,
260            len_before: bcx.iconst(isize_type, 0),
261            len_offset: 0,
262            bcx,
263
264            bytecode,
265            inst_entries,
266            current_inst: usize::MAX,
267
268            incoming_dynamic_jumps: Vec::new(),
269            dynamic_jump_table,
270
271            incoming_failures: Vec::new(),
272            failure_block: Some(failure_block),
273            incoming_returns: Vec::new(),
274            return_block: Some(return_block),
275
276            resume_kind: ResumeKind::Indexes,
277            resume_blocks: Vec::new(),
278            suspend_blocks: Vec::new(),
279            suspend_block,
280
281            builtins,
282        };
283
284        // We store the stack length if requested or necessary due to the bytecode.
285        let stack_length_observable = config.inspect_stack_length || bytecode.may_suspend();
286
287        // Add debug assertions for the parameters.
288        if config.debug_assertions {
289            fx.pointer_panic_with_bool(
290                config.gas_metering,
291                gas_ptr,
292                "gas pointer",
293                "gas metering is enabled",
294            );
295            fx.pointer_panic_with_bool(
296                !config.local_stack,
297                sp_arg,
298                "stack pointer",
299                "local stack is disabled",
300            );
301            fx.pointer_panic_with_bool(
302                stack_length_observable,
303                stack_len_arg,
304                "stack length pointer",
305                if config.inspect_stack_length {
306                    "stack length inspection is enabled"
307                } else {
308                    "bytecode suspends execution"
309                },
310            );
311            fx.pointer_panic_with_bool(true, env, "env pointer", "");
312            fx.pointer_panic_with_bool(true, contract, "contract pointer", "");
313            fx.pointer_panic_with_bool(true, ecx, "EVM context pointer", "");
314        }
315
316        // The bytecode is guaranteed to have at least one instruction.
317        let first_inst_block = fx.inst_entries[0];
318        let post_entry_block = fx.bcx.create_block_after(entry_block, "entry.post");
319        let resume_block = fx.bcx.create_block_after(post_entry_block, "resume");
320        fx.bcx.br(post_entry_block);
321
322        // Translate individual instructions into their respective blocks.
323        for (inst, _) in bytecode.iter_insts() {
324            fx.translate_inst(inst)?;
325        }
326
327        // Finalize the dynamic jump table.
328        fx.bcx.switch_to_block(unreachable_block);
329        fx.bcx.unreachable();
330        if bytecode.has_dynamic_jumps() {
331            fx.bcx.switch_to_block(fx.dynamic_jump_table);
332            // TODO: Manually reduce to i32?
333            let jumpdests = bytecode.iter_insts().filter(|(_, data)| data.opcode == op::JUMPDEST);
334            // let max_pc =
335            //     jumpdests.clone().map(|(_, data)| data.pc).next_back().expect("no jumpdests");
336            let targets = jumpdests
337                .map(|(inst, data)| (data.pc as u64, fx.inst_entries[inst]))
338                .collect::<Vec<_>>();
339            let index = fx.bcx.phi(fx.word_type, &fx.incoming_dynamic_jumps);
340            // let target =
341            //     fx.bcx.create_block_after(fx.dynamic_jump_table, "dynamic_jump_table.contd");
342            // let overflow = fx.bcx.icmp_imm(IntCC::UnsignedGreaterThan, index, max_pc as i64);
343            // fx.bcx.brif(overflow, default, target);
344
345            // fx.bcx.switch_to_block(target);
346            // let index = fx.bcx.ireduce(i32_type, index);
347            fx.add_invalid_jump();
348            fx.bcx.switch(index, return_block, &targets, true);
349        } else {
350            // No dynamic jumps.
351            debug_assert!(fx.incoming_dynamic_jumps.is_empty());
352            fx.bcx.switch_to_block(fx.dynamic_jump_table);
353            fx.bcx.unreachable();
354        }
355
356        // Finalize the suspend and resume blocks. Must come before the return block.
357        // Also here is where the stack length is initialized.
358        let load_len_at_start = |fx: &mut Self| {
359            // Loaded from args only for the config.
360            if config.inspect_stack_length {
361                let stack_len = fx.bcx.load(fx.isize_type, stack_len_arg, "stack_len");
362                fx.stack_len.store(&mut fx.bcx, stack_len);
363            } else {
364                fx.stack_len.store_imm(&mut fx.bcx, 0);
365            }
366        };
367        let generate_resume = bytecode.may_suspend();
368        if generate_resume {
369            let get_ecx_resume_at_ptr = |fx: &mut Self| {
370                fx.get_field(
371                    fx.ecx,
372                    mem::offset_of!(EvmContext<'_>, resume_at),
373                    "ecx.resume_at.addr",
374                )
375            };
376
377            let kind = fx.resume_kind;
378            let resume_ty = match kind {
379                ResumeKind::Blocks => fx.ptr_type,
380                ResumeKind::Indexes => fx.isize_type,
381            };
382
383            // Resume block: load the `resume_at` value and switch to the corresponding block.
384            // Invalid values are treated as unreachable.
385            {
386                // Special-case the no resume case to load 0 into the length if possible.
387                let no_resume_block = fx.bcx.create_block_after(resume_block, "no_resume");
388
389                fx.bcx.switch_to_block(post_entry_block);
390                let resume_at = get_ecx_resume_at_ptr(&mut fx);
391                let resume_at = fx.bcx.load(resume_ty, resume_at, "ecx.resume_at");
392                let no_resume = match kind {
393                    ResumeKind::Blocks => fx.bcx.is_null(resume_at),
394                    ResumeKind::Indexes => fx.bcx.icmp_imm(IntCC::Equal, resume_at, 0),
395                };
396                fx.bcx.brif(no_resume, no_resume_block, resume_block);
397
398                fx.bcx.switch_to_block(no_resume_block);
399                load_len_at_start(&mut fx);
400                fx.bcx.br(first_inst_block);
401
402                // Dispatch to the resume block.
403                fx.bcx.switch_to_block(resume_block);
404                let stack_len = fx.bcx.load(fx.isize_type, stack_len_arg, "stack_len");
405                fx.stack_len.store(&mut fx.bcx, stack_len);
406                match kind {
407                    ResumeKind::Blocks => {
408                        fx.bcx.br_indirect(resume_at, &fx.resume_blocks);
409                    }
410                    ResumeKind::Indexes => {
411                        let default = fx.bcx.create_block_after(resume_block, "resume_invalid");
412                        fx.bcx.switch_to_block(default);
413                        fx.call_panic("invalid `resume_at` value");
414
415                        fx.bcx.switch_to_block(resume_block);
416                        let targets = fx
417                            .resume_blocks
418                            .iter()
419                            .enumerate()
420                            .map(|(i, b)| (i as u64 + 1, *b))
421                            .collect::<Vec<_>>();
422                        fx.bcx.switch(resume_at, default, &targets, true);
423                    }
424                }
425            }
426
427            // Suspend block: store the `resume_at` value and return `CallOrCreate`.
428            {
429                fx.bcx.switch_to_block(fx.suspend_block);
430                let resume_value = fx.bcx.phi(resume_ty, &fx.suspend_blocks);
431                let resume_at = get_ecx_resume_at_ptr(&mut fx);
432                fx.bcx.store(resume_value, resume_at);
433
434                fx.build_return_imm(InstructionResult::CallOrCreate);
435            }
436        } else {
437            debug_assert!(fx.resume_blocks.is_empty());
438            debug_assert!(fx.suspend_blocks.is_empty());
439
440            fx.bcx.switch_to_block(post_entry_block);
441            load_len_at_start(&mut fx);
442            fx.bcx.br(first_inst_block);
443
444            fx.bcx.switch_to_block(resume_block);
445            fx.bcx.unreachable();
446            fx.bcx.switch_to_block(fx.suspend_block);
447            fx.bcx.unreachable();
448        }
449
450        // Finalize the failure block.
451        fx.bcx.switch_to_block(fx.failure_block.unwrap());
452        if !fx.incoming_failures.is_empty() {
453            let failure_value = fx.bcx.phi(fx.i8_type, &fx.incoming_failures);
454            fx.bcx.set_current_block_cold();
455            fx.build_return(failure_value);
456        } else {
457            fx.bcx.unreachable();
458        }
459
460        // Finalize the return block.
461        fx.bcx.switch_to_block(fx.return_block.unwrap());
462        if !fx.incoming_returns.is_empty() {
463            let return_value = fx.bcx.phi(fx.i8_type, &fx.incoming_returns);
464            if stack_length_observable {
465                fx.save_stack_len();
466            }
467            fx.bcx.ret(&[return_value]);
468        } else {
469            fx.bcx.unreachable();
470        }
471
472        fx.bcx.seal_all_blocks();
473
474        Ok(())
475    }

477    #[instrument(level = "debug", skip_all, fields(inst = %self.bytecode.inst(inst).to_op()))]
478    fn translate_inst(&mut self, inst: Inst) -> Result<()> {
479        self.current_inst = inst;
480        let data = self.bytecode.inst(inst);
481        let opcode = data.opcode;
482        let entry_block = self.inst_entries[inst];
483        self.bcx.switch_to_block(entry_block);
484
485        let is_eof = self.bytecode.is_eof();
486        let is_eof_enabled = self.bytecode.spec_id.is_enabled_in(SpecId::OSAKA);
487        if is_eof {
488            ensure!(is_eof_enabled, "EOF bytecode in non-EOF spec");
489        }
490
491        // self.call_printf(format_printf!("{}\n", self.op_block_name("")), &[]);
492
493        let branch_to_next_opcode = |this: &mut Self| {
494            debug_assert!(
495                !this.bytecode.is_instr_diverging(inst),
496                "attempted to branch to next instruction in a diverging instruction: {data:?}",
497            );
498            if let Some(next) = this.inst_entries.get(inst + 1) {
499                this.bcx.br(*next);
500            }
501        };
502        // Currently a noop.
503        // let epilogue = |this: &mut Self| {
504        //     this.bcx.seal_block(entry_block);
505        // };
506
507        /// Makes sure to run cleanup code and return.
508        /// Use `no_branch` to skip the branch to the next opcode.
509        /// Use `build` to build the return instruction and skip the branch.
510        macro_rules! goto_return {
511            (no_branch $($comment:expr)?) => {
512                $(
513                    if self.config.comments {
514                        self.add_comment($comment);
515                    }
516                )?
517                // epilogue(self);
518                return Ok(());
519            };
520            (build $ret:expr) => {{
521                self.build_return_imm($ret);
522                goto_return!(no_branch);
523            }};
524            (fail $ret:expr) => {{
525                self.build_fail_imm($ret);
526                goto_return!(no_branch);
527            }};
528            ($($comment:expr)?) => {
529                branch_to_next_opcode(self);
530                goto_return!(no_branch $($comment)?);
531            };
532        }
533
534        // Assert that we already skipped the block.
535        debug_assert!(!data.flags.contains(InstFlags::DEAD_CODE));
536
537        #[cfg(test)]
538        if opcode == crate::TEST_SUSPEND {
539            self.suspend();
540            goto_return!(no_branch);
541        }
542
543        // This is a compile error because it should've been validated as per EOF.
544        if is_eof_enabled && is_eof {
545            if let Some(info) = OPCODE_INFO_JUMPTABLE[opcode as usize] {
546                ensure!(
547                    !info.is_disabled_in_eof(),
548                    "disabled opcode in EOF bytecode: {}",
549                    data.to_op_in(self.bytecode),
550                );
551            }
552        }
553
554        // Revm doesn't consider spec ID when checking for EOF-only opcodes,
555        // so don't check for `is_eof_enabled`.
556        if !is_eof && data.flags.contains(InstFlags::EOF_ONLY) {
557            // Match Revm output.
558            let ret = if opcode == op::RETURNCONTRACT {
559                InstructionResult::ReturnContractInNotInitEOF
560            } else {
561                InstructionResult::EOFOpcodeDisabledInLegacy
562            };
563            goto_return!(fail ret);
564        }
565
566        // Disabled instructions don't pay gas.
567        if data.flags.contains(InstFlags::DISABLED) {
568            goto_return!(fail InstructionResult::NotActivated);
569        }
570        if data.flags.contains(InstFlags::UNKNOWN) {
571            ensure!(!is_eof, "Unknown opcode in EOF bytecode: {data:?}");
572            goto_return!(fail InstructionResult::OpcodeNotFound);
573        }
574
575        if is_eof {
576            if let Some(info) = OPCODE_INFO_JUMPTABLE[opcode as usize] {
577                ensure!(!info.is_disabled_in_eof(), "Disabled opcode in EOF bytecode: {data:?}");
578            }
579        }
580
581        // Pay static gas for the current section.
582        self.gas_cost_imm(data.section.gas_cost as u64);
583
584        if data.flags.contains(InstFlags::SKIP_LOGIC) {
585            goto_return!("skipped");
586        }
587
588        // Reset the stack length offset for this instruction.
589        self.len_offset = 0;
590        self.len_before = self.stack_len.load(&mut self.bcx, "stack_len");
591
592        // Check stack length for the current section.
593        // Skip doing this for EOF bytecode, as it is done at deploy time.
594        if !is_eof && self.config.stack_bound_checks {
595            let inp = data.section.inputs;
596            let diff = data.section.max_growth as i64;
597
598            if diff > revmc_context::EvmStack::CAPACITY as i64 {
599                goto_return!(fail InstructionResult::StackOverflow);
600            }
601
602            let underflow = |this: &mut Self| {
603                debug_assert!(inp > 0);
604                this.bcx.icmp_imm(IntCC::UnsignedLessThan, this.len_before, inp as i64)
605            };
606            let overflow = |this: &mut Self| {
607                debug_assert!(diff > 0 && diff <= STACK_CAP as i64);
608                this.bcx.icmp_imm(
609                    IntCC::UnsignedGreaterThan,
610                    this.len_before,
611                    STACK_CAP as i64 - diff,
612                )
613            };
614
615            let may_underflow = inp > 0;
616            let may_overflow = diff > 0;
617            if may_underflow && may_overflow {
618                let underflow = underflow(self);
619                let overflow = overflow(self);
620                let cond = self.bcx.bitor(underflow, overflow);
621                let ret = {
622                    let under =
623                        self.bcx.iconst(self.i8_type, InstructionResult::StackUnderflow as i64);
624                    let over =
625                        self.bcx.iconst(self.i8_type, InstructionResult::StackOverflow as i64);
626                    self.bcx.select(underflow, under, over)
627                };
628                let target = self.build_check_inner(true, cond, ret);
629                self.bcx.switch_to_block(target);
630            } else if may_underflow {
631                let cond = underflow(self);
632                self.build_check(cond, InstructionResult::StackUnderflow);
633            } else if may_overflow {
634                let cond = overflow(self);
635                self.build_check(cond, InstructionResult::StackOverflow);
636            }
637        }

639        // Update the stack length for this instruction.
640        {
641            let (inp, out) = data.stack_io();
642            let diff = out as i64 - inp as i64;
643            if diff != 0 {
644                let mut diff = diff;
645                // HACK: For now all opcodes that suspend (minus the test one, which does not reach
646                // here) return exactly one value. This value is pushed onto the stack by the
647                // caller, so we don't account for it here.
648                if data.may_suspend(is_eof) {
649                    diff -= 1;
650                }
651                let len_changed = self.bcx.iadd_imm(self.len_before, diff);
652                self.stack_len.store(&mut self.bcx, len_changed);
653            }
654        }

665        macro_rules! binop {
666            ($op:ident) => {{
667                let [a, b] = self.popn();
668                let r = self.bcx.$op(a, b);
669                self.push(r);
670            }};
671            (@shift $op:ident, | $value:ident, $shift:ident | $default:expr) => {{
672                let [$shift, $value] = self.popn();
673                let r = self.bcx.$op($value, $shift);
674                let overflow = self.bcx.icmp_imm(IntCC::UnsignedGreaterThan, $shift, 255);
675                let default = $default;
676                let r = self.bcx.select(overflow, default, r);
677                self.push(r);
678            }};
679            (@if_not_zero $op:ident) => {{
680                // TODO: `select` might not have the same semantics in all backends.
681                let [a, b] = self.popn();
682                let b_is_zero = self.bcx.icmp_imm(IntCC::Equal, b, 0);
683                let zero = self.bcx.iconst_256(U256::ZERO);
684                let op_result = self.bcx.$op(a, b);
685                let r = self.bcx.select(b_is_zero, zero, op_result);
686                self.push(r);
687            }};
688        }

690        macro_rules! field {
691            // Gets the pointer to a field.
692            ($field:ident; @get $($paths:path),*; $($spec:tt).*) => {
693                self.get_field(self.$field, 0 $(+ mem::offset_of!($paths, $spec))*, stringify!($field.$($spec).*.addr))
694            };
695            // Gets and loads the pointer to a field.
696            // The value is loaded as a native-endian 256-bit integer.
697            // `@[endian]` is the endianness of the value. If native, omit it.
698            ($field:ident; @load $(@[endian = $endian:tt])? $ty:expr, $($paths:path),*; $($spec:tt).*) => {{
699                let ptr = field!($field; @get $($paths),*; $($spec).*);
700                #[allow(unused_mut)]
701                let mut value = self.bcx.load($ty, ptr, stringify!($field.$($spec).*));
702                $(
703                    if !cfg!(target_endian = $endian) {
704                        value = self.bcx.bswap(value);
705                    }
706                )?
707                value
708            }};
709            // Gets, loads, extends (if necessary), and pushes the value of a field to the stack.
710            // `@[endian]` is the endianness of the value. If native, omit it.
711            ($field:ident; @push $(@[endian = $endian:tt])? $ty:expr, $($rest:tt)*) => {{
712                let mut value = field!($field; @load $(@[endian = $endian])? $ty, $($rest)*);
713                if self.bcx.type_bit_width($ty) < 256 {
714                    value = self.bcx.zext(self.word_type, value);
715                }
716                self.push(value);
717            }};
718        }

1162            }
1163            op::CREATE2 => {
1164                self.create_common(CreateKind::Create2);
1165                goto_return!(no_branch);
1166            }
1167
1168            op::RETURNDATALOAD => {
1169                let sp = self.sp_after_inputs();
1170                let _ = self.call_builtin(Builtin::ReturnDataLoad, &[self.ecx, sp]);
1171            }
1172            op::EXTCALL => {
1173                self.ext_call_common(ExtCallKind::Call);
1174                goto_return!(no_branch);
1175            }
1176            op::EXTDELEGATECALL => {
1177                self.ext_call_common(ExtCallKind::DelegateCall);
1178                goto_return!(no_branch);
1179            }
1180            op::STATICCALL => {
1181                self.call_common(CallKind::StaticCall);
1182                goto_return!(no_branch);
1183            }
1184            op::EXTSTATICCALL => {
1185                self.ext_call_common(ExtCallKind::StaticCall);
1186                goto_return!(no_branch);
1187            }
1188
1189            op::REVERT => {
1190                self.return_common(InstructionResult::Revert);
1191                goto_return!(no_branch);
1192            }
1193            op::INVALID => goto_return!(fail InstructionResult::InvalidFEOpcode),
1194            op::SELFDESTRUCT => {
1195                let sp = self.sp_after_inputs();
1196                let spec_id = self.const_spec_id();
1197                self.call_fallible_builtin(Builtin::SelfDestruct, &[self.ecx, sp, spec_id]);

1199            }

1201            _ => unreachable!("unimplemented instruction: {data:?}"),
1202        }
1203

1205    }

1333    /// Builds a `CALLF` or `JUMPF` instruction.
1334    fn callf_common(&mut self, imm: &[u8], is_jumpf: bool) {
1335        let op_name = if is_jumpf { "JUMPF" } else { "CALLF" };
1336
1337        let idx = u16::from_be_bytes(imm.try_into().unwrap()) as usize;
1338
1339        // Check stack max height.
1340        let types = self
1341            .expect_eof()
1342            .body
1343            .types_section
1344            .get(idx)
1345            .unwrap_or_else(|| panic!("{op_name} section {idx}: types not found"));
1346        let max_height = types.max_stack_size - types.inputs as u16;
1347        let mut max_len = self.len_before();
1348        if max_height != 0 {
1349            max_len = self.bcx.iadd_imm(max_len, max_height as i64);
1350        }
1351        let cond = self.bcx.icmp_imm(IntCC::UnsignedGreaterThan, max_len, STACK_CAP as i64);
1352        self.build_check(cond, InstructionResult::StackOverflow);
1353
1354        // Push the return address to the function stack.
1355        let next_block = self.inst_entries[self.current_inst + 1];
1356        if is_jumpf {
1357            self.func_stack_set(idx);
1358        } else {
1359            let value = match self.bcx.block_addr(next_block) {
1360                Some(addr) => addr,
1361                None => todo!(),
1362            };
1363            self.call_func_stack_push(value, idx);
1364        }
1365
1366        let inst = self.bytecode.eof_section_inst(idx);
1367        self.bcx.br(self.inst_entries[inst]);
1368    }

1391    /// Suspend execution, storing the resume point in the context.
1392    fn suspend(&mut self) {
1393        // Register the next instruction as the resume block.
1394        let idx = self.resume_blocks.len();
1395        let value = self.add_resume_at(self.inst_entries[self.current_inst + 1]);
1396
1397        // Register the current block as the suspend block.
1398        let value = match value {
1399            Some(value) => value,
1400            None => self.bcx.iconst(self.isize_type, idx as i64 + 1),
1401        };
1402        self.suspend_blocks.push((value, self.bcx.current_block().unwrap()));
1403
1404        // Branch to the suspend block.
1405        self.bcx.br(self.suspend_block);
1406    }

1513        // This can overflow the gas counters, which has to be adjusted for after the call.

1515        let (res, overflow) = self.bcx.usub_overflow(gas_remaining, cost);

1560    #[must_use]
1561    fn build_check_inner(
1562        &mut self,
1563        is_failure: bool,
1564        cond: B::Value,
1565        ret: B::Value,
1566    ) -> B::BasicBlock {
1567        let current_block = self.current_block();
1568        let target = self.create_block_after(current_block, "contd");
1569
1570        let return_block = if let Some(return_block) = self.return_block {
1571            self.incoming_returns.push((ret, current_block));
1572            return_block
1573        } else {
1574            self.create_block_after(target, "return")
1575        };
1576        let then_block = if is_failure { return_block } else { target };
1577        let else_block = if is_failure { target } else { return_block };
1578        self.bcx.brif_cold(cond, then_block, else_block, is_failure);
1579
1580        if self.return_block.is_none() {
1581            self.bcx.switch_to_block(return_block);
1582            self.bcx.ret(&[ret]);
1583        }
1584
1585        target
1586    }

1761    fn build_byte(&mut self) {
1762        let index = self.bcx.fn_param(0);
1763        let value = self.bcx.fn_param(1);
1764
1765        let cond = self.bcx.icmp_imm(IntCC::UnsignedLessThan, index, 32);
1766        let byte = {
1767            // (value >> (31 - index) * 8) & 0xFF
1768            let thirty_one = self.bcx.iconst_256(U256::from(31));
1769            let shift = self.bcx.isub(thirty_one, index);
1770            let shift = self.bcx.imul_imm(shift, 8);
1771            let shifted = self.bcx.ushr(value, shift);
1772            let mask = self.bcx.iconst_256(U256::from(0xFF));
1773            self.bcx.bitand(shifted, mask)
1774        };
1775        let zero = self.bcx.iconst_256(U256::ZERO);
1776        let r = self.bcx.select(cond, byte, zero);
1777
1778        self.bcx.ret(&[r]);
1779    }

1785    /// Builds: `fn signextend(ext: u256, x: u256) -> u256`
1786    fn build_signextend(&mut self) {
1787        // From the yellow paper:
1788        /*
1789        let [ext, x] = stack.pop();
1790        let t = 256 - 8 * (ext + 1);
1791        let mut result = x;
1792        result[..t] = [x[t]; t]; // Index by bits.
1793        */
1794
1795        let ext = self.bcx.fn_param(0);
1796        let x = self.bcx.fn_param(1);
1797
1798        // For 31 we also don't need to do anything.
1799        let might_do_something = self.bcx.icmp_imm(IntCC::UnsignedLessThan, ext, 31);
1800        let r = self.bcx.lazy_select(
1801            might_do_something,
1802            self.bcx.type_int(256),
1803            |bcx| {
1804                // Adapted from revm: https://github.com/bluealloy/revm/blob/fda371f73aba2c30a83c639608be78145fd1123b/crates/interpreter/src/instructions/arithmetic.rs#L89
1805                // let bit_index = 8 * ext + 7;
1806                // let bit = (x >> bit_index) & 1 != 0;
1807                // let mask = (1 << bit_index) - 1;
1808                // let r = if bit { x | !mask } else { *x & mask };
1809
1810                // let bit_index = 8 * ext + 7;
1811                let bit_index = bcx.imul_imm(ext, 8);
1812                let bit_index = bcx.iadd_imm(bit_index, 7);
1813
1814                // let bit = (x >> bit_index) & 1 != 0;
1815                let one = bcx.iconst_256(U256::from(1));
1816                let bit = bcx.ushr(x, bit_index);
1817                let bit = bcx.bitand(bit, one);
1818                let bit = bcx.icmp_imm(IntCC::NotEqual, bit, 0);
1819
1820                // let mask = (1 << bit_index) - 1;
1821                let mask = bcx.ishl(one, bit_index);
1822                let mask = bcx.isub_imm(mask, 1);
1823
1824                // let r = if bit { x | !mask } else { *x & mask };
1825                let not_mask = bcx.bitnot(mask);
1826                let sext = bcx.bitor(x, not_mask);
1827                let zext = bcx.bitand(x, mask);
1828                bcx.select(bit, sext, zext)
1829            },
1830            |_bcx| x,
1831        );
1832        self.bcx.ret(&[r]);
1833    }

1942    /// Builds:
1943    /// - `Load` => `fn mload(offset: u256, out: ptr, ecx: ptr) -> InstructionResult`
1944    /// - `Store` => `fn mstore(offset: u256, value: u256, ecx: ptr) -> InstructionResult`
1945    /// - `Store8` => `fn mstore(offset: u256, value: u8, ecx: ptr) -> InstructionResult`
1946    fn build_mem_op(&mut self, kind: MemOpKind) {
1947        let is_load = matches!(kind, MemOpKind::Load);
1948        let ptr_args = if is_load { &[1, 2][..] } else { &[2][..] };
1949        for &ptr_arg in ptr_args {
1950            for attr in default_attrs::for_ref() {
1951                self.bcx.add_function_attribute(
1952                    None,
1953                    attr,
1954                    FunctionAttributeLocation::Param(ptr_arg),
1955                )
1956            }
1957        }
1958        if is_load {
1959            self.bcx.add_function_attribute(
1960                None,
1961                Attribute::WriteOnly,
1962                FunctionAttributeLocation::Param(1),
1963            );
1964        }
1965
1966        let offset = self.bcx.fn_param(0);
1967        let value = self.bcx.fn_param(1);
1968        let ecx = self.bcx.fn_param(2);
1969
1970        let memory_ptr = {
1971            let memory_ptr_ptr =
1972                self.get_field(ecx, mem::offset_of!(EvmContext<'_>, memory), "ecx.memory.addr");
1973            self.bcx.load(self.ptr_type, memory_ptr_ptr, "ecx.memory")
1974        };
1975
1976        let memory_buffer_offset = mem::offset_of!(pf::SharedMemory, buffer);
1977        let len_ptr = self.get_field(
1978            memory_ptr,
1979            memory_buffer_offset + mem::offset_of!(pf::Vec<u8>, len),
1980            "ecx.memory.len.addr",
1981        );
1982        let sm_len = self.bcx.load(self.isize_type, len_ptr, "ecx.memory.len");
1983
1984        // `memory.len() = memory.buffer.len() - memory.last_checkpoint`
1985        // `new_size = offset + len`
1986        // `if new_size > memory.len() { resize_memory(new_size) }`
1987        let last_checkpoint = {
1988            let ptr = self.get_field(
1989                memory_ptr,
1990                mem::offset_of!(pf::SharedMemory, last_checkpoint),
1991                "ecx.memory.last_checkpoint.addr",
1992            );
1993            self.bcx.load(self.isize_type, ptr, "ecx.memory.last_checkpoint")
1994        };
1995        let buffer_len = self.bcx.isub(sm_len, last_checkpoint);
1996        let max_isize = ((1u128 << self.bcx.type_bit_width(self.isize_type)) - 1u128) as u64;
1997        let max_isize_u256 = self.bcx.iconst_256(U256::from(max_isize));
1998        let max_isize = self.bcx.uconst(self.isize_type, max_isize);
1999        let offset_too_big = self.bcx.icmp(IntCC::UnsignedGreaterThan, offset, max_isize_u256);
2000        let offset = self.bcx.ireduce(self.isize_type, offset);
2001        let (new_size, new_size_overflow) = {
2002            let slot_size = match kind {
2003                MemOpKind::Load | MemOpKind::Store => 32,
2004                MemOpKind::Store8 => 1,
2005            };
2006            let slot_size = self.bcx.iconst(self.isize_type, slot_size as i64);
2007            self.bcx.uadd_overflow(offset, slot_size)
2008        };
2009        let new_size_overflow = self.bcx.bitor(offset_too_big, new_size_overflow);
2010        let new_size = self.bcx.select(new_size_overflow, max_isize, new_size);
2011        let cond = self.bcx.icmp(IntCC::UnsignedGreaterThan, new_size, buffer_len);
2012
2013        let resize = self.bcx.create_block("resize");
2014        let cont = self.bcx.create_block("contd");
2015        self.bcx.brif_cold(cond, resize, cont, true);
2016
2017        self.bcx.switch_to_block(resize);
2018        self.call_fallible_builtin(Builtin::ResizeMemory, &[ecx, new_size]);
2019        self.bcx.br(cont);
2020
2021        // `ecx.memory.buffer[last_checkpoint + offset..]`
2022        // Implemented as `ecx.memory.buffer[last_checkpoint..][offset..]`
2023        self.bcx.switch_to_block(cont);
2024        let shared_buffer_ptr = {
2025            let ptr = self.get_field(
2026                memory_ptr,
2027                memory_buffer_offset + mem::offset_of!(pf::Vec<u8>, ptr),
2028                "ecx.memory.buffer.ptr.shared.addr",
2029            );
2030            self.bcx.load(self.ptr_type, ptr, "ecx.memory.buffer.ptr.shared")
2031        };
2032        let buffer_ptr = self.bcx.gep(
2033            self.i8_type,
2034            shared_buffer_ptr,
2035            &[last_checkpoint],
2036            "ecx.memory.buffer.ptr",
2037        );
2038        let slot = self.bcx.gep(self.i8_type, buffer_ptr, &[offset], "slot");
2039        match kind {
2040            MemOpKind::Load => {
2041                let loaded = self.bcx.load_unaligned(self.word_type, slot, "slot.value");
2042                let loaded =
2043                    if cfg!(target_endian = "little") { self.bcx.bswap(loaded) } else { loaded };
2044                self.bcx.store(loaded, value);
2045            }
2046            MemOpKind::Store | MemOpKind::Store8 => {
2047                let value = if matches!(kind, MemOpKind::Store) && cfg!(target_endian = "little") {
2048                    self.bcx.bswap(value)
2049                } else {
2050                    value
2051                };
2052                self.bcx.store_unaligned(value, slot);
2053            }
2054        }
2055
2056        let cont = self.const_continue();
2057        self.bcx.ret(&[cont]);
2058    }

2259        });
