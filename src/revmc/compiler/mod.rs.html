34/// EVM bytecode compiler.
35///
36/// This currently represents one single-threaded IR context and module, which can be used to
37/// compile multiple functions as JIT or AOT.
38///
39/// Functions can be incrementally added with [`translate`], and then either written to an object
40/// file with [`write_object`] when in AOT mode, or JIT-compiled with [`jit_function`].
41///
42/// Performing either of these operations finalizes the module, and no more functions can be added
43/// afterwards until [`clear`] is called, which will reset the module to its initial state.
44///
45/// [`translate`]: EvmCompiler::translate
46/// [`write_object`]: EvmCompiler::write_object
47/// [`jit_function`]: EvmCompiler::jit_function
48/// [`clear`]: EvmCompiler::clear
49#[allow(missing_debug_implementations)]
50pub struct EvmCompiler<B: Backend> {
51    name: Option<String>,
52    backend: B,
53    out_dir: Option<PathBuf>,
54    config: FcxConfig,
55    builtins: Builtins<B>,
56
57    dump_assembly: bool,
58    dump_unopt_assembly: bool,
59
60    finalized: bool,
61}

229    /// Translates the given EVM bytecode into an internal function.
230    ///
231    /// NOTE: `name` must be unique for each function, as it is used as the name of the final
232    /// symbol.
233    pub fn translate<'a>(
234        &mut self,
235        name: &str,
236        input: impl Into<EvmCompilerInput<'a>>,
237        spec_id: SpecId,
238    ) -> Result<B::FuncId> {
239        ensure!(cfg!(target_endian = "little"), "only little-endian is supported");
240        ensure!(!self.finalized, "cannot compile more functions after finalizing the module");
241        let bytecode = self.parse(input.into(), spec_id)?;
242        self.translate_inner(name, &bytecode)
243    }

245    /// (JIT) Compiles the given EVM bytecode into a JIT function.
246    ///
247    /// See [`translate`](Self::translate) for more information.
248    ///
249    /// # Safety
250    ///
251    /// The returned function pointer is owned by the module, and must not be called after the
252    /// module is cleared or the function is freed.
253    pub unsafe fn jit<'a>(
254        &mut self,
255        name: &str,
256        bytecode: impl Into<EvmCompilerInput<'a>>,
257        spec_id: SpecId,
258    ) -> Result<EvmCompilerFn> {
259        let id = self.translate(name, bytecode.into(), spec_id)?;
260        unsafe { self.jit_function(id) }
261    }
262
263    /// (JIT) Finalizes the module and JITs the given function.
264    ///
265    /// # Safety
266    ///
267    /// The returned function pointer is owned by the module, and must not be called after the
268    /// module is cleared or the function is freed.
269    pub unsafe fn jit_function(&mut self, id: B::FuncId) -> Result<EvmCompilerFn> {
270        ensure!(self.is_jit(), "cannot JIT functions during AOT compilation");
271        self.finalize()?;
272        let addr = self.backend.jit_function(id)?;
273        debug_assert!(addr != 0);
274        Ok(EvmCompilerFn::new(unsafe { std::mem::transmute::<usize, RawEvmCompilerFn>(addr) }))
275    }

280        let mut writer = io::BufWriter::new(file);
281        self.write_object(&mut writer)?;
282        writer.flush()?;
283        Ok(())
284    }

286    /// (AOT) Finalizes the module and writes the compiled object to the given writer.

293    /// (JIT) Frees the memory associated with a single function.
294    ///
295    /// Note that this will not reset the state of the internal module even if all functions are
296    /// freed with this function. Use [`clear`] to reset the module.
297    ///
298    /// [`clear`]: EvmCompiler::clear
299    ///
300    /// # Safety
301    ///
302    /// Because this function invalidates any pointers retrieved from the corresponding module, it
303    /// should only be used when none of the functions from that module are currently executing and
304    /// none of the `fn` pointers are called afterwards.
305    pub unsafe fn free_function(&mut self, id: B::FuncId) -> Result<()> {
306        self.backend.free_function(id)
307    }
308
309    /// Frees all functions and resets the state of the internal module, allowing for new functions
310    /// to be compiled.
311    ///
312    /// # Safety
313    ///
314    /// Because this function invalidates any pointers retrieved from the corresponding module, it
315    /// should only be used when none of the functions from that module are currently executing and
316    /// none of the `fn` pointers are called afterwards.
317    pub unsafe fn clear(&mut self) -> Result<()> {
318        self.builtins.clear();
319        self.finalized = false;
320        self.backend.free_all_functions()
321    }

323    /// Parses the given EVM bytecode. Not public API.
324    #[doc(hidden)] // Not public API.
325    pub fn parse<'a>(
326        &mut self,
327        input: EvmCompilerInput<'a>,
328        spec_id: SpecId,
329    ) -> Result<Bytecode<'a>> {
330        let bytecode;
331        let eof;
332        match input {
333            EvmCompilerInput::Code(code) => {
334                bytecode = code;
335                if spec_id.is_enabled_in(SpecId::OSAKA) && code.starts_with(&EOF_MAGIC_BYTES) {
336                    eof = Some(Cow::Owned(Eof::decode(Bytes::copy_from_slice(code))?));
337                } else {
338                    eof = None;
339                }
340            }
341            EvmCompilerInput::Eof(e) => {
342                bytecode = &e.raw[..];
343                eof = Some(Cow::Borrowed(e));
344            }
345        }
346        if let Some(eof) = &eof {
347            self.do_validate_eof(eof)?;
348        }
349
350        let mut bytecode = Bytecode::new(bytecode, eof, spec_id);
351        bytecode.analyze()?;
352        if let Some(dump_dir) = &self.dump_dir() {
353            Self::dump_bytecode(dump_dir, &bytecode)?;
354        }
355        Ok(bytecode)
356    }
357
358    fn do_validate_eof(&self, eof: &Eof) -> Result<()> {
359        if !self.config.validate_eof {
360            return Ok(());
361        }
362        revm_interpreter::analysis::validate_eof_inner(eof, None).map_err(|e| match e {
363            revm_interpreter::analysis::EofError::Decode(e) => e.into(),
364            revm_interpreter::analysis::EofError::Validation(e) => {
365                eyre!("validation error: {e:?}")
366            }
367        })
368    }

370    #[instrument(name = "translate", level = "debug", skip_all)]
371    fn translate_inner(&mut self, name: &str, bytecode: &Bytecode<'_>) -> Result<B::FuncId> {
372        ensure!(self.backend.function_name_is_unique(name), "function name `{name}` is not unique");
373        let linkage = Linkage::Public;
374        let (bcx, id) = Self::make_builder(&mut self.backend, &self.config, name, linkage)?;
375        FunctionCx::translate(bcx, self.config, &mut self.builtins, bytecode)?;
376        Ok(id)
377    }

379    #[instrument(level = "debug", skip_all)]
380    fn finalize(&mut self) -> Result<()> {
381        if self.finalized {
382            return Ok(());
383        }
384        self.finalized = true;
385
386        if let Some(dump_dir) = &self.dump_dir() {
387            let path = dump_dir.join("unopt").with_extension(self.backend.ir_extension());
388            self.dump_ir(&path)?;
389
390            // Dump IR before verifying for better debugging.
391            self.verify_module()?;
392
393            if self.dump_assembly && self.dump_unopt_assembly {
394                let path = dump_dir.join("unopt.s");
395                self.dump_disasm(&path)?;
396            }
397        } else {
398            self.verify_module()?;
399        }
400
401        self.optimize_module()?;
402
403        if let Some(dump_dir) = &self.dump_dir() {
404            let path = dump_dir.join("opt").with_extension(self.backend.ir_extension());
405            self.dump_ir(&path)?;
406
407            if self.dump_assembly {
408                let path = dump_dir.join("opt.s");
409                self.dump_disasm(&path)?;
410            }
411        }
412
413        Ok(())
414    }

416    #[instrument(level = "debug", skip_all)]
417    fn make_builder<'a>(
418        backend: &'a mut B,
419        config: &FcxConfig,
420        name: &str,
421        linkage: Linkage,
422    ) -> Result<(B::Builder<'a>, B::FuncId)> {
423        fn size_align<T>(i: usize) -> (usize, usize, usize) {
424            (i, mem::size_of::<T>(), mem::align_of::<T>())
425        }
426
427        let i8 = backend.type_int(8);
428        let ptr = backend.type_ptr();
429        let (ret, params, param_names, ptr_attrs) = (
430            Some(i8),
431            &[ptr, ptr, ptr, ptr, ptr, ptr],
432            &[
433                "arg.gas.addr",
434                "arg.stack.addr",
435                "arg.stack_len.addr",
436                "arg.env.addr",
437                "arg.contract.addr",
438                "arg.ecx.addr",
439            ],
440            &[
441                size_align::<Gas>(0),
442                size_align::<EvmStack>(1),
443                size_align::<usize>(2),
444                size_align::<Env>(3),
445                size_align::<Contract>(4),
446                size_align::<EvmContext<'_>>(5),
447            ],
448        );
449        debug_assert_eq!(params.len(), param_names.len());
450        let (mut bcx, id) = backend.build_function(name, ret, params, param_names, linkage)?;
451
452        // Function attributes.
453        let function_attributes = default_attrs::for_fn()
454            .chain(config.frame_pointers.then_some(Attribute::AllFramePointers))
455            // We can unwind in panics, which are present only in debug assertions.
456            .chain((!config.debug_assertions).then_some(Attribute::NoUnwind));
457        for attr in function_attributes {
458            bcx.add_function_attribute(None, attr, FunctionAttributeLocation::Function);
459        }
460
461        // Pointer argument attributes.
462        if !config.debug_assertions {
463            for &(i, size, align) in ptr_attrs {
464                let attrs = default_attrs::for_sized_ptr((size, align))
465                    // `Gas` is aliased in `EvmContext`.
466                    .chain((i != 0).then_some(Attribute::NoAlias));
467                for attr in attrs {
468                    let loc = FunctionAttributeLocation::Param(i as _);
469                    bcx.add_function_attribute(None, attr, loc);
470                }
471            }
472        }
473
474        Ok((bcx, id))
475    }

498    fn dump_bytecode(dump_dir: &Path, bytecode: &Bytecode<'_>) -> Result<()> {
499        {
500            let file = fs::File::create(dump_dir.join("bytecode.txt"))?;
501            let mut writer = io::BufWriter::new(file);
502            write!(writer, "{bytecode}")?;
503            writer.flush()?;
504        }
505
506        {
507            let file = fs::File::create(dump_dir.join("bytecode.dbg.txt"))?;
508            let mut writer = io::BufWriter::new(file);
509            writeln!(writer, "{bytecode:#?}")?;
510            writer.flush()?;
511        }
512
513        Ok(())
514    }
515
516    fn dump_dir(&self) -> Option<PathBuf> {
517        let mut dump_dir = self.out_dir.clone()?;
518        if let Some(name) = &self.name {
519            dump_dir.push(name.replace(char::is_whitespace, "_"));
520        }
521        if !dump_dir.exists() {
522            let _ = fs::create_dir_all(&dump_dir);
523        }
524        Some(dump_dir)
525    }

528/// [`EvmCompiler`] input.
529#[allow(missing_debug_implementations)]
530pub enum EvmCompilerInput<'a> {
531    /// EVM bytecode. Can also be raw EOF code, which will be parsed.
532    Code(&'a [u8]),
533    /// Already-parsed EOF container.
534    Eof(&'a Eof),
535}
536
537impl<'a> From<&'a [u8]> for EvmCompilerInput<'a> {
538    fn from(code: &'a [u8]) -> Self {
539        EvmCompilerInput::Code(code)
540    }
541}
542
543impl<'a> From<&'a Vec<u8>> for EvmCompilerInput<'a> {
544    fn from(code: &'a Vec<u8>) -> Self {
545        EvmCompilerInput::Code(code)
546    }
547}
548
549impl<'a> From<&'a Bytes> for EvmCompilerInput<'a> {
550    fn from(code: &'a Bytes) -> Self {
551        EvmCompilerInput::Code(code)
552    }
553}
554
555impl<'a> From<&'a Eof> for EvmCompilerInput<'a> {
556    fn from(eof: &'a Eof) -> Self {
557        EvmCompilerInput::Eof(eof)
558    }
559}

561#[allow(dead_code)]
562mod default_attrs {
563    use revmc_backend::Attribute;
564
565    pub(crate) fn for_fn() -> impl Iterator<Item = Attribute> {
566        [
567            Attribute::WillReturn,      // Always returns.
568            Attribute::NoSync,          // No thread synchronization.
569            Attribute::NativeTargetCpu, // Optimization.
570            Attribute::Speculatable,    // No undefined behavior.
571            Attribute::NoRecurse,       // Revm is not recursive.
572        ]
573        .into_iter()
574    }
575
576    pub(crate) fn for_param() -> impl Iterator<Item = Attribute> {
577        [Attribute::NoUndef].into_iter()
578    }
579
580    pub(crate) fn for_ptr() -> impl Iterator<Item = Attribute> {
581        for_param().chain([Attribute::NoCapture])
582    }
583
584    pub(crate) fn for_sized_ptr((size, align): (usize, usize)) -> impl Iterator<Item = Attribute> {
585        for_ptr().chain([Attribute::Dereferenceable(size as u64), Attribute::Align(align as u64)])
586    }
587
588    pub(crate) fn for_ptr_t<T>() -> impl Iterator<Item = Attribute> {
589        for_sized_ptr(size_align::<T>())
590    }
591
592    pub(crate) fn for_ref() -> impl Iterator<Item = Attribute> {
593        for_ptr().chain([Attribute::NonNull, Attribute::NoAlias])
594    }
595
596    pub(crate) fn for_sized_ref((size, align): (usize, usize)) -> impl Iterator<Item = Attribute> {
597        for_ref().chain([Attribute::Dereferenceable(size as u64), Attribute::Align(align as u64)])
598    }
599
600    pub(crate) fn for_ref_t<T>() -> impl Iterator<Item = Attribute> {
601        for_sized_ref(size_align::<T>())
602    }
603
604    pub(crate) fn size_align<T>() -> (usize, usize) {
605        (std::mem::size_of::<T>(), std::mem::align_of::<T>())
606    }
