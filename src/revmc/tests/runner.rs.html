26impl<'a> arbitrary::Arbitrary<'a> for TestCase<'a> {
27    fn arbitrary(u: &mut arbitrary::Unstructured<'a>) -> arbitrary::Result<Self> {
28        let is_eof_enabled = u.arbitrary::<bool>()?;
29        let spec_id_range = if is_eof_enabled {
30            SpecId::OSAKA as u8..=SpecId::OSAKA as u8
31        } else {
32            0..=(SpecId::OSAKA as u8 - 1)
33        };
34        let spec_id = SpecId::try_from_u8(u.int_in_range(spec_id_range)?).unwrap_or(DEF_SPEC);
35
36        let mut bytecode: &'a [u8] = u.arbitrary()?;
37        if is_eof_enabled && !bytecode.starts_with(&EOF_MAGIC_BYTES) && u.arbitrary()? {
38            let code = eof_sections_unchecked(&[bytecode]);
39            if revm_interpreter::analysis::validate_eof(&code).is_ok() {
40                static mut STORAGE: Bytes = Bytes::new();
41                unsafe {
42                    STORAGE = code.raw;
43                    bytecode = &STORAGE[..];
44                }
45            }
46        }
47
48        Ok(Self::what_interpreter_says(bytecode, spec_id))
49    }
50}

318pub fn with_evm_context<F: FnOnce(&mut EvmContext<'_>, &mut EvmStack, &mut usize) -> R, R>(
319    bytecode: &[u8],
320    f: F,
321) -> R {
322    let contract = Contract {
323        input: Bytes::from_static(DEF_CD),
324        bytecode: revm_interpreter::analysis::to_analysed(revm_primitives::Bytecode::new_raw(
325            Bytes::copy_from_slice(bytecode),
326        )),
327        hash: None,
328        bytecode_address: None,
329        target_address: DEF_ADDR,
330        caller: DEF_CALLER,
331        call_value: DEF_VALUE,
332    };
333
334    let mut interpreter = revm_interpreter::Interpreter::new(contract, DEF_GAS_LIMIT, false);
335    interpreter.return_data_buffer = Bytes::from_static(DEF_RD);
336
337    let mut host = TestHost::new();
338
339    let (mut ecx, stack, stack_len) =
340        EvmContext::from_interpreter_with_stack(&mut interpreter, &mut host);
341    f(&mut ecx, stack, stack_len)
342}

368pub fn run_test_case<B: Backend>(test_case: &TestCase<'_>, compiler: &mut EvmCompiler<B>) {
369    let TestCase { bytecode, spec_id, .. } = *test_case;
370    compiler.inspect_stack_length(true);
371    // Done manually in `fn eof` and friends.
372    compiler.validate_eof(false);
373    // compiler.debug_assertions(false);
374    let f = unsafe { compiler.jit("test", bytecode, spec_id) }.unwrap();
375    run_compiled_test_case(test_case, f);
376}
377
378fn run_compiled_test_case(test_case: &TestCase<'_>, f: EvmCompilerFn) {
379    let TestCase {
380        bytecode,
381        spec_id,
382        modify_ecx,
383        expected_return,
384        expected_stack,
385        expected_memory,
386        expected_gas,
387        ref expected_next_action,
388        assert_host,
389        assert_ecx,
390    } = *test_case;
391
392    let is_eof_enabled = spec_id.is_enabled_in(SpecId::OSAKA);
393
394    if !is_eof_enabled && bytecode.starts_with(&primitives::EOF_MAGIC_BYTES) {
395        panic!("EOF is not enabled in the current spec, forgot to set `spec_id`?");
396    }
397
398    with_evm_context(bytecode, |ecx, stack, stack_len| {
399        if let Some(modify_ecx) = modify_ecx {
400            modify_ecx(ecx);
401        }
402
403        if !cfg!(feature = "__fuzzing") && is_eof_enabled && !ecx.contract.bytecode.is_eof() {
404            eprintln!("!!! WARNING: running legacy code under EOF !!!");
405        }
406
407        // Interpreter.
408        let table = spec_to_generic!(test_case.spec_id, op::make_instruction_table::<_, SPEC>());
409        let mut interpreter = ecx.to_interpreter(Default::default());
410        let memory = interpreter.take_memory();
411        let mut int_host = TestHost::new();
412
413        let interpreter_action = interpreter.run(memory, &table, &mut int_host);
414
415        let mut expected_return = expected_return;
416        if expected_return == RETURN_WHAT_INTERPRETER_SAYS {
417            expected_return = interpreter.instruction_result;
418        } else {
419            assert_eq!(
420                interpreter.instruction_result, expected_return,
421                "interpreter return value mismatch"
422            );
423        }
424
425        let mut expected_stack = expected_stack;
426        if expected_stack == STACK_WHAT_INTERPRETER_SAYS {
427            expected_stack = interpreter.stack.data();
428        } else {
429            assert_eq!(interpreter.stack.data(), expected_stack, "interpreter stack mismatch");
430        }
431
432        let mut expected_memory = expected_memory;
433        if expected_memory == MEMORY_WHAT_INTERPRETER_SAYS {
434            expected_memory = interpreter.shared_memory.context_memory();
435        } else {
436            assert_eq!(
437                MemDisplay(interpreter.shared_memory.context_memory()),
438                MemDisplay(expected_memory),
439                "interpreter memory mismatch"
440            );
441        }
442
443        let mut expected_gas = expected_gas;
444        if expected_gas == GAS_WHAT_INTERPRETER_SAYS {
445            expected_gas = interpreter.gas.spent();
446        } else {
447            assert_eq!(interpreter.gas.spent(), expected_gas, "interpreter gas mismatch");
448        }
449
450        // This is what the interpreter returns when the internal action is None in `run`.
451        let default_action = InterpreterAction::Return {
452            result: InterpreterResult {
453                result: interpreter.instruction_result,
454                output: Bytes::new(),
455                gas: interpreter.gas,
456            },
457        };
458        let mut expected_next_action = expected_next_action;
459        if *expected_next_action == ACTION_WHAT_INTERPRETER_SAYS {
460            expected_next_action = &interpreter_action;
461        } else {
462            if expected_next_action.is_none() {
463                expected_next_action = &default_action;
464            }
465            assert_actions(&interpreter_action, expected_next_action);
466        }
467
468        if let Some(assert_host) = assert_host {
469            assert_host(&int_host);
470        }
471
472        let actual_return = unsafe { f.call(Some(stack), Some(stack_len), ecx) };
473
474        if matches!(
475            actual_return,
476            // We can have a stack overflow/underflow before other error codes due to sections.
477            |InstructionResult::StackOverflow| InstructionResult::StackUnderflow
478            // Any OOG is equivalent. We skip `InvalidOperand` sometimes.
479            | InstructionResult::OutOfGas | InstructionResult::MemoryOOG | InstructionResult::InvalidOperandOOG
480        ) {
481            assert_eq!(
482                actual_return.is_error(),
483                expected_return.is_error(),
484                "return value mismatch: {actual_return:?} != {expected_return:?}"
485            );
486        } else {
487            assert_eq!(actual_return, expected_return, "return value mismatch");
488        }
489
490        let actual_stack =
491            stack.as_slice().iter().take(*stack_len).map(|x| x.to_u256()).collect::<Vec<_>>();
492
493        // On EVM halt all available gas is consumed, so resulting stack, memory, and gas do not
494        // matter. We do less work than the interpreter by bailing out earlier due to sections.
495        if !actual_return.is_error() {
496            assert_eq!(actual_stack, *expected_stack, "stack mismatch");
497
498            assert_eq!(
499                MemDisplay(ecx.memory.context_memory()),
500                MemDisplay(expected_memory),
501                "interpreter memory mismatch"
502            );
503
504            assert_eq!(ecx.gas.spent(), expected_gas, "gas mismatch");
505        }
506
507        let actual_next_action =
508            if ecx.next_action.is_none() { &default_action } else { &*ecx.next_action };
509        assert_actions(actual_next_action, expected_next_action);
510
511        if let Some(_assert_host) = assert_host {
512            #[cfg(not(feature = "__fuzzing"))]
513            _assert_host(ecx.host.downcast_ref().unwrap());
514        }
515
516        if let Some(assert_ecx) = assert_ecx {
517            assert_ecx(ecx);
518        }
519    });
520}

524impl fmt::Debug for MemDisplay<'_> {

527        f.debug_list().entries(chunks).finish()
528    }

531#[track_caller]
532fn assert_actions(actual: &InterpreterAction, expected: &InterpreterAction) {
533    match (actual, expected) {
534        (
535            InterpreterAction::Return { result },
536            InterpreterAction::Return { result: expected_result },
537        ) => {
538            assert_eq!(result.result, expected_result.result, "result mismatch");
539            assert_eq!(result.output, expected_result.output, "result output mismatch");
540            if expected_result.gas.limit() != GAS_WHAT_INTERPRETER_SAYS {
541                assert_eq!(result.gas.spent(), expected_result.gas.spent(), "result gas mismatch");
542            }
543        }
544        (a, b) => assert_eq!(a, b, "next action mismatch"),
545    }
