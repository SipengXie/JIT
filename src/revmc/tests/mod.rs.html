1280#[track_caller]
1281fn eof_sections(code: &[&[u8]]) -> Bytes {
1282    let eof = eof_sections_unchecked(code);
1283    match revm_interpreter::analysis::validate_eof_inner(&eof, None) {
1284        Ok(()) => {}
1285        Err(EofError::Decode(e)) => panic!("{e}"),
1286        Err(EofError::Validation(e)) => match e {
1287            EofValidationError::UnknownOpcode
1288                if code.iter().any(|code| code.contains(&TEST_SUSPEND)) => {}
1289            EofValidationError::InvalidTypesSection | EofValidationError::MaxStackMismatch => {}
1290            e => panic!("validation error: {e:?}"),
1291        },
1292    }
1293    eof.raw
1294}

1301fn eof_body(code: &[&[u8]], containers: Vec<Bytes>) -> primitives::eof::EofBody {
1302    revm_primitives::eof::EofBody {
1303        types_section: {
1304            let mut types =
1305                vec![primitives::eof::TypesSection { inputs: 0, outputs: 0x80, max_stack_size: 0 }];
1306            for _ in 1..code.len() {
1307                types.push(primitives::eof::TypesSection {
1308                    inputs: 0,
1309                    outputs: 0,
1310                    max_stack_size: 0,
1311                });
1312            }
1313            types
1314        },
1315        code_section: code.iter().copied().map(Bytes::copy_from_slice).collect(),
1316        container_section: containers,
1317        data_section: Bytes::from_static(DEF_DATA),
1318        is_data_filled: true,
1319    }
1320}
