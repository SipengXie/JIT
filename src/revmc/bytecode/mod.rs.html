25/// An EVM instruction is a high level internal representation of an EVM opcode.
26///
27/// This is an index into [`Bytecode`] instructions.
28///
29/// Also known as `ic`, or instruction counter; not to be confused with SSA `inst`s.
30pub(crate) type Inst = usize;
31
32/// EVM bytecode.
33#[doc(hidden)] // Not public API.
34pub struct Bytecode<'a> {
35    /// The original bytecode slice.
36    pub(crate) code: &'a [u8],
37    /// The parsed EOF container, if any.
38    eof: Option<Cow<'a, Eof>>,
39    /// The instructions.
40    insts: Vec<InstData>,
41    /// `JUMPDEST` opcode map. `jumpdests[pc]` is `true` if `code[pc] == op::JUMPDEST`.
42    jumpdests: BitVec,
43    /// The [`SpecId`].
44    pub(crate) spec_id: SpecId,
45    /// Whether the bytecode contains dynamic jumps. Always false in EOF.
46    has_dynamic_jumps: bool,
47    /// Whether the bytecode may suspend execution.
48    may_suspend: bool,
49    /// Mapping from program counter to instruction.
50    pc_to_inst: FxHashMap<u32, u32>,
51    /// Mapping from EOF code section index to the list of instructions that call it.
52    eof_called_by: Vec<Vec<Inst>>,
53}

56    #[instrument(name = "new_bytecode", level = "debug", skip_all)]
57    pub(crate) fn new(mut code: &'a [u8], eof: Option<Cow<'a, Eof>>, spec_id: SpecId) -> Self {
58        if let Some(eof) = &eof {
59            code = unsafe {
60                std::slice::from_raw_parts(
61                    eof.body.code_section.first().unwrap().as_ptr(),
62                    eof.header.sum_code_sizes,
63                )
64            };
65        }
66
67        let is_eof = eof.is_some();
68
69        let mut insts = Vec::with_capacity(code.len() + 8);
70        // JUMPDEST analysis is not done in EOF.
71        let mut jumpdests = if is_eof { BitVec::new() } else { BitVec::repeat(false, code.len()) };
72        let mut pc_to_inst = FxHashMap::with_capacity_and_hasher(code.len(), Default::default());
73        let op_infos = op_info_map(spec_id);
74        for (inst, (pc, Opcode { opcode, immediate: _ })) in
75            OpcodesIter::new(code, spec_id).with_pc().enumerate()
76        {
77            pc_to_inst.insert(pc as u32, inst as u32);
78
79            if !is_eof && opcode == op::JUMPDEST {
80                jumpdests.set(pc, true)
81            }
82
83            let data = 0;
84
85            let mut flags = InstFlags::empty();
86            let info = op_infos[opcode as usize];
87            if info.is_unknown() {
88                flags |= InstFlags::UNKNOWN;
89            }
90            if info.is_disabled() {
91                flags |= InstFlags::DISABLED;
92            }
93            if info.is_eof_only() {
94                flags |= InstFlags::EOF_ONLY;
95            }
96            let base_gas = info.base_gas();
97
98            let section = Section::default();
99
100            insts.push(InstData { opcode, flags, base_gas, data, pc: pc as u32, section });
101        }
102
103        let mut bytecode = Self {
104            code,
105            eof,
106            insts,
107            jumpdests,
108            spec_id,
109            has_dynamic_jumps: false,
110            may_suspend: false,
111            pc_to_inst,
112            eof_called_by: vec![],
113        };
114
115        // Pad code to ensure there is at least one diverging instruction.
116        // EOF enforces this, so there is no need to pad it ourselves.
117        if !is_eof && bytecode.insts.last().map_or(true, |last| !last.is_diverging(false)) {
118            bytecode.insts.push(InstData::new(op::STOP));
119        }
120
121        bytecode
122    }

186    /// Runs a list of analysis passes on the instructions.
187    #[instrument(level = "debug", skip_all)]
188    pub(crate) fn analyze(&mut self) -> Result<()> {
189        if !self.is_eof() {
190            self.static_jump_analysis();
191            // NOTE: `mark_dead_code` must run after `static_jump_analysis` as it can mark
192            // unreachable `JUMPDEST`s as dead code.
193            self.mark_dead_code();
194        }
195
196        self.calc_may_suspend();
197
198        if self.is_eof() {
199            self.calc_eof_called_by()?;
200            self.eof_mark_jumpdests();
201        }
202
203        self.construct_sections();
204
205        Ok(())
206    }

208    /// Mark `PUSH<N>` followed by `JUMP[I]` as `STATIC_JUMP` and resolve the target.
209    #[instrument(name = "sj", level = "debug", skip_all)]
210    fn static_jump_analysis(&mut self) {
211        debug_assert!(!self.is_eof());
212
213        for jump_inst in 0..self.insts.len() {
214            let jump = &self.insts[jump_inst];
215            let Some(push_inst) = jump_inst.checked_sub(1) else {
216                if jump.is_legacy_jump() {
217                    trace!(jump_inst, target=?None::<()>, "found jump");
218                    self.has_dynamic_jumps = true;
219                }
220                continue;
221            };
222
223            let push = &self.insts[push_inst];
224            if !(push.is_push() && jump.is_legacy_jump()) {
225                if jump.is_legacy_jump() {
226                    trace!(jump_inst, target=?None::<()>, "found jump");
227                    self.has_dynamic_jumps = true;
228                }
229                continue;
230            }
231
232            let imm_opt = self.get_imm(push);
233            if push.opcode != op::PUSH0 && imm_opt.is_none() {
234                continue;
235            }
236            let imm = imm_opt.unwrap_or(&[]);
237            self.insts[jump_inst].flags |= InstFlags::STATIC_JUMP;
238
239            const USIZE_SIZE: usize = std::mem::size_of::<usize>();
240            if imm.len() > USIZE_SIZE {
241                trace!(jump_inst, "jump target too large");
242                self.insts[jump_inst].flags |= InstFlags::INVALID_JUMP;
243                continue;
244            }
245
246            let mut padded = [0; USIZE_SIZE];
247            padded[USIZE_SIZE - imm.len()..].copy_from_slice(imm);
248            let target_pc = usize::from_be_bytes(padded);
249            if !self.is_valid_jump(target_pc) {
250                trace!(jump_inst, target_pc, "invalid jump target");
251                self.insts[jump_inst].flags |= InstFlags::INVALID_JUMP;
252                continue;
253            }
254
255            self.insts[push_inst].flags |= InstFlags::SKIP_LOGIC;
256            let target = self.pc_to_inst(target_pc);
257
258            // Mark the `JUMPDEST` as reachable.
259            debug_assert_eq!(
260                self.insts[target],
261                op::JUMPDEST,
262                "is_valid_jump returned true for non-JUMPDEST: \
263                 jump_inst={jump_inst} target_pc={target_pc} target={target}",
264            );
265            self.insts[target].data = 1;
266
267            // Set the target on the `JUMP` instruction.
268            trace!(jump_inst, target, "found jump");
269            self.insts[jump_inst].data = target as u32;
270        }
271    }

273    /// Mark `RJUMP*` targets with `EOF_JUMPDEST` flag.
274    #[instrument(name = "eof_sj", level = "debug", skip_all)]
275    fn eof_mark_jumpdests(&mut self) {
276        debug_assert!(self.is_eof());
277
278        for inst in 0..self.insts.len() {
279            let data = self.inst(inst);
280            if data.is_eof_jump() {
281                for (_, pc) in self.iter_rjump_targets(data) {
282                    let target_inst = self.pc_to_inst(pc);
283                    self.inst_mut(target_inst).flags |= InstFlags::EOF_JUMPDEST;
284                }
285            }
286        }
287    }

289    /// Mark unreachable instructions as `DEAD_CODE` to not generate any code for them.
290    ///
291    /// This pass is technically unnecessary as the backend will very likely optimize any
292    /// unreachable code that we generate, but this is trivial for us to do and significantly speeds
293    /// up code generation.
294    ///
295    /// We can simply mark all instructions that are between diverging instructions and
296    /// `JUMPDEST`s.
297    #[instrument(name = "dce", level = "debug", skip_all)]
298    fn mark_dead_code(&mut self) {
299        debug_assert!(!self.is_eof());
300
301        let mut iter = self.insts.iter_mut().enumerate();
302        while let Some((i, data)) = iter.next() {
303            if data.is_diverging(false) {
304                let mut end = i;
305                for (j, data) in &mut iter {
306                    end = j;
307                    if data.is_reachable_jumpdest(false, self.has_dynamic_jumps) {
308                        break;
309                    }
310                    data.flags |= InstFlags::DEAD_CODE;
311                }
312                let start = i + 1;
313                if end > start {
314                    debug!("found dead code: {start}..{end}");
315                }
316            }
317        }
318    }

320    /// Calculates whether the bytecode suspend suspend execution.
321    ///
322    /// This can only happen if the bytecode contains `*CALL*` or `*CREATE*` instructions.
323    #[instrument(name = "suspend", level = "debug", skip_all)]
324    fn calc_may_suspend(&mut self) {
325        let is_eof = self.is_eof();
326        let may_suspend = self.iter_insts().any(|(_, data)| data.may_suspend(is_eof));
327        self.may_suspend = may_suspend;
328    }

330    /// Constructs the sections in the bytecode.
331    #[instrument(name = "sections", level = "debug", skip_all)]
332    fn construct_sections(&mut self) {
333        let mut analysis = SectionAnalysis::default();
334        for inst in 0..self.insts.len() {
335            if !self.inst(inst).is_dead_code() {
336                analysis.process(self, inst);
337            }
338        }
339        analysis.finish(self);
340    }

342    /// Constructs the sections in the bytecode.
343    #[instrument(name = "sections", level = "debug", skip_all)]
344    #[cfg(any())]
345    fn construct_sections_default(&mut self) {
346        for inst in &mut self.insts {
347            let (inp, out) = inst.stack_io();
348            let stack_diff = out as i16 - inp as i16;
349            inst.section =
350                Section { gas_cost: inst.base_gas as _, inputs: inp as _, max_growth: stack_diff }
351        }
352    }

354    /// Calculates the list of instructions that call each EOF section.
355    ///
356    /// This is done to compute the `indirectbr` destinations of `RETF` instructions.
357    #[instrument(name = "eof_called_by", level = "debug", skip_all)]
358    fn calc_eof_called_by(&mut self) -> Result<()> {
359        let code_sections_len = self.expect_eof().body.code_section.len();
360        if code_sections_len <= 1 {
361            return Ok(());
362        }
363
364        // First, collect all `CALLF` targets.
365        let mut eof_called_by = vec![Vec::new(); code_sections_len];
366        for (inst, data) in self.iter_all_insts() {
367            if data.opcode == op::CALLF {
368                let imm = self.get_imm(data).unwrap();
369                let target_section = u16::from_be_bytes(imm.try_into().unwrap()) as usize;
370                eof_called_by[target_section].push(inst);
371            }
372        }
373
374        // Then, propagate `JUMPF` calls.
375        const MAX_ITERATIONS: usize = 32;
376        let mut any_progress = true;
377        let mut i = 0usize;
378        let first_section_inst = self.eof_section_inst(1);
379        while any_progress && i < MAX_ITERATIONS {
380            any_progress = false;
381
382            for (_inst, data) in self.iter_all_insts().skip(first_section_inst) {
383                if data.opcode == op::JUMPF {
384                    let source_section = self.pc_to_eof_section(data.pc as usize);
385                    debug_assert!(source_section != 0);
386
387                    let imm = self.get_imm(data).unwrap();
388                    let target_section = u16::from_be_bytes(imm.try_into().unwrap()) as usize;
389
390                    let (source_section, target_section) =
391                        get_two_mut(&mut eof_called_by, source_section, target_section);
392
393                    for &source_call in &*source_section {
394                        if !target_section.contains(&source_call) {
395                            any_progress = true;
396                            target_section.push(source_call);
397                        }
398                    }
399                }
400            }
401
402            i += 1;
403        }
404        // TODO: Is this actually reachable?
405        // If so, we should remove this error and handle this case properly by making all `CALLF`
406        // reachable.
407        ensure!(i < MAX_ITERATIONS, "`calc_eof_called_by` did not converge");
408        self.eof_called_by = eof_called_by;
409        Ok(())
410    }

417    /// Returns the immediate value of the given instruction data, if any.
418    /// Returns `None` if out of bounds too.
419    pub(crate) fn get_imm(&self, data: &InstData) -> Option<&'a [u8]> {
420        let mut imm_len = data.imm_len() as usize;
421        if imm_len == 0 {
422            return None;
423        }
424        let start = data.pc as usize + 1;
425        if data.opcode == op::RJUMPV {
426            imm_len += (*self.code.get(start)? as usize + 1) * 2;
427        }
428        self.code.get(start..start + imm_len)
429    }

479    /// Returns the program counter of the given EOF section index.
480    pub(crate) fn eof_section_pc(&self, section: usize) -> Inst {
481        let code = &self.expect_eof().body.code_section;
482        let first = code.first().unwrap().as_ptr();
483        let section_ptr = code[section].as_ptr();
484        section_ptr as usize - first as usize
485    }

492    pub(crate) fn pc_to_eof_section(&self, pc: usize) -> usize {
493        (0..self.expect_eof().body.code_section.len())
494            .rev()
495            .find(|&section| pc >= self.eof_section_pc(section))
496            .unwrap()
497    }

499    /// Iterates over the index and `RJUMP` target instructions of the given instruction.
500    pub(crate) fn iter_rjump_target_insts(
501        &self,
502        data: &InstData,
503    ) -> impl Iterator<Item = (usize, Inst)> + '_ {
504        let from = data.pc;
505        self.iter_rjump_targets(data).map(move |(i, pc)| {
506            self.eof_assert_jump_in_bounds(from as usize, pc);
507            (i, self.pc_to_inst(pc))
508        })
509    }

511    /// Iterates over the index and `RJUMP` target PCs of the given instruction.
512    pub(crate) fn iter_rjump_targets(
513        &self,
514        data: &InstData,
515    ) -> impl Iterator<Item = (usize, usize)> + 'a {
516        let opcode = data.opcode;
517        let pc = data.pc;
518        let imm = self.get_imm(data).unwrap();
519
520        debug_assert!(InstData::new(opcode).is_eof_jump());
521        if matches!(opcode, op::RJUMP | op::RJUMPI) {
522            let offset = i16::from_be_bytes(imm.try_into().unwrap());
523            let base_pc = pc + 3;
524            let target_pc = (base_pc as usize).wrapping_add(offset as usize);
525            return Either::Left(std::iter::once((0, target_pc)));
526        }
527
528        let max_index = imm[0] as usize;
529        let base_pc = pc + 2 + (max_index as u32 + 1) * 2;
530        Either::Right(imm[1..].chunks(2).enumerate().map(move |(i, chunk)| {
531            debug_assert!(i <= max_index);
532            debug_assert_eq!(chunk.len(), 2);
533            let offset = i16::from_be_bytes(chunk.try_into().unwrap());
534            let target_pc = (base_pc as usize).wrapping_add(offset as usize);
535            (i, target_pc)
536        }))
537    }

555    /// Returns the name for a basic block.
556    pub(crate) fn op_block_name(&self, mut inst: usize, name: &str) -> String {
557        use std::fmt::Write;
558
559        if inst == usize::MAX {
560            return format!("entry.{name}");
561        }
562        let mut section = None;
563        let data = self.inst(inst);
564        if self.is_eof() {
565            let section_index = self.pc_to_eof_section(data.pc as usize);
566            section = Some(section_index);
567            inst -= self.eof_section_inst(section_index);
568        }
569
570        let mut s = String::new();
571        if let Some(section) = section {
572            let _ = write!(s, "S{section}.");
573        }
574        let _ = write!(s, "OP{inst}.{}", data.to_op());
575        if !name.is_empty() {
576            let _ = write!(s, ".{name}");
577        }
578        s
579    }

582impl fmt::Display for Bytecode<'_> {
583    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
584        let header = format!("{:^6} | {:^6} | {:^80} | {}", "ic", "pc", "opcode", "instruction");
585        writeln!(f, "{header}")?;
586        writeln!(f, "{}", "-".repeat(header.len()))?;
587        for (inst, (pc, opcode)) in self.opcodes().with_pc().enumerate() {
588            let data = self.inst(inst);
589            let opcode = opcode.to_string();
590            writeln!(f, "{inst:>6} | {pc:>6} | {opcode:<80} | {data:?}")?;
591        }
592        Ok(())
593    }
594}

610/// A single instruction in the bytecode.
611#[derive(Clone, Default)]
612pub(crate) struct InstData {
613    /// The opcode byte.
614    pub(crate) opcode: u8,
615    /// Flags.
616    pub(crate) flags: InstFlags,
617    /// The base gas cost of the opcode.
618    ///
619    /// This may not be the final/full gas cost of the opcode as it may also have a dynamic cost.
620    base_gas: u16,
621    /// Instruction-specific data:
622    /// - if the instruction has immediate data, this is a packed offset+length into the bytecode;
623    /// - `JUMP{,I} && STATIC_JUMP in kind`: the jump target, `Instr`;
624    /// - `JUMPDEST`: `1` if the jump destination is reachable, `0` otherwise;
625    /// - otherwise: no meaning.
626    pub(crate) data: u32,
627    /// The program counter, meaning `code[pc]` is this instruction's opcode.
628    pub(crate) pc: u32,
629    /// The section this instruction belongs to.
630    pub(crate) section: Section,
631}
632
633impl PartialEq<u8> for InstData {
634    #[inline]
635    fn eq(&self, other: &u8) -> bool {
636        self.opcode == *other
637    }
638}
639
640impl PartialEq<InstData> for u8 {
641    #[inline]
642    fn eq(&self, other: &InstData) -> bool {
643        *self == other.opcode
644    }
645}
646
647impl fmt::Debug for InstData {
648    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
649        f.debug_struct("InstData")
650            .field("opcode", &self.to_op())
651            .field("flags", &format_args!("{:?}", self.flags))
652            .field("data", &self.data)
653            .field("pc", &self.pc)
654            .field("section", &self.section)
655            .finish()
656    }
657}
658
659impl InstData {
660    /// Creates a new instruction data with the given opcode byte.
661    /// Note that this may not be a valid instruction.
662    #[inline]
663    fn new(opcode: u8) -> Self {
664        Self { opcode, ..Default::default() }
665    }
666
667    /// Returns the length of the immediate data of this instruction.
668    #[inline]
669    pub(crate) const fn imm_len(&self) -> u8 {
670        min_imm_len(self.opcode)
671    }
672
673    /// Returns the number of input and output stack elements of this instruction.
674    #[inline]
675    pub(crate) fn stack_io(&self) -> (u8, u8) {
676        let (mut inp, out) = stack_io(self.opcode);
677        if self.is_legacy_static_jump()
678            && !(self.opcode == op::JUMPI && self.flags.contains(InstFlags::INVALID_JUMP))
679        {
680            inp -= 1;
681        }
682        (inp, out)
683    }
684
685    /// Converts this instruction to a raw opcode. Note that the immediate data is not resolved.
686    #[inline]
687    pub(crate) const fn to_op(&self) -> Opcode<'static> {
688        Opcode { opcode: self.opcode, immediate: None }
689    }
690
691    /// Converts this instruction to a raw opcode in the given bytecode.
692    #[inline]
693    #[allow(dead_code)]
694    pub(crate) fn to_op_in<'a>(&self, bytecode: &Bytecode<'a>) -> Opcode<'a> {
695        Opcode { opcode: self.opcode, immediate: bytecode.get_imm(self) }
696    }
697
698    /// Returns `true` if this instruction is a push instruction.
699    #[inline]
700    pub(crate) fn is_push(&self) -> bool {
701        matches!(self.opcode, op::PUSH0..=op::PUSH32)
702    }
703
704    /// Returns `true` if this instruction is a jump instruction.
705    #[inline]
706    fn is_jump(&self, is_eof: bool) -> bool {
707        if is_eof {
708            self.is_eof_jump()
709        } else {
710            self.is_legacy_jump()
711        }
712    }
713
714    /// Returns `true` if this instruction is an EOF jump instruction (`RJUMP`/`RJUMPI`/`RJUMPV`).
715    #[inline]
716    fn is_eof_jump(&self) -> bool {
717        matches!(self.opcode, op::RJUMP | op::RJUMPI | op::RJUMPV)
718    }
719
720    /// Returns `true` if this instruction is a legacy jump instruction (`JUMP`/`JUMPI`).
721    #[inline]
722    pub(crate) fn is_legacy_jump(&self) -> bool {
723        matches!(self.opcode, op::JUMP | op::JUMPI)
724    }
725
726    /// Returns `true` if this instruction is a legacy jump instruction (`JUMP`/`JUMPI`), and the
727    /// target known statically.
728    #[inline]
729    pub(crate) fn is_legacy_static_jump(&self) -> bool {
730        self.is_legacy_jump() && self.flags.contains(InstFlags::STATIC_JUMP)
731    }
732
733    /// Returns `true` if this instruction is a `JUMPDEST`.
734    #[inline]
735    pub(crate) const fn is_jumpdest(&self) -> bool {
736        self.opcode == op::JUMPDEST
737    }
738
739    /// Returns `true` if this instruction is a reachable `JUMPDEST`.
740    #[inline]
741    pub(crate) const fn is_reachable_jumpdest(
742        &self,
743        is_eof: bool,
744        has_dynamic_jumps: bool,
745    ) -> bool {
746        if is_eof {
747            self.flags.contains(InstFlags::EOF_JUMPDEST)
748        } else {
749            self.is_jumpdest() && (has_dynamic_jumps || self.data == 1)
750        }
751    }
752
753    /// Returns `true` if this instruction is dead code.
754    pub(crate) fn is_dead_code(&self) -> bool {
755        self.flags.contains(InstFlags::DEAD_CODE)
756    }
757
758    /// Returns `true` if this instruction requires to know `gasleft()`.
759    /// Note that this does not include CALL and CREATE.
760    #[inline]
761    pub(crate) fn requires_gasleft(&self, spec_id: SpecId) -> bool {
762        // For SSTORE, see `revm_interpreter::gas::sstore_cost`.
763        self.opcode == op::GAS
764            || (self.opcode == op::SSTORE && spec_id.is_enabled_in(SpecId::ISTANBUL))
765    }
766
767    /// Returns `true` if we know that this instruction will branch or stop execution.
768    #[inline]
769    pub(crate) fn is_branching(&self, is_eof: bool) -> bool {
770        self.is_jump(is_eof) || self.is_diverging(is_eof)
771    }
772
773    /// Returns `true` if we know that this instruction will stop execution.
774    #[inline]
775    pub(crate) fn is_diverging(&self, is_eof: bool) -> bool {
776        #[cfg(test)]
777        if self.opcode == TEST_SUSPEND {
778            return false;
779        }
780
781        (self.opcode == op::JUMP && self.flags.contains(InstFlags::INVALID_JUMP))
782            || self.flags.contains(InstFlags::DISABLED)
783            || self.flags.contains(InstFlags::UNKNOWN)
784            || matches!(self.opcode, op::STOP | op::RETURN | op::REVERT | op::INVALID)
785            || (!is_eof && matches!(self.opcode, op::SELFDESTRUCT))
786            || (is_eof && matches!(self.opcode, op::JUMPF | op::RETF | op::RETURNCONTRACT))
787    }
788
789    /// Returns `true` if this instruction may suspend execution.
790    #[inline]
791    pub(crate) const fn may_suspend(&self, is_eof: bool) -> bool {
792        #[cfg(test)]
793        if self.opcode == TEST_SUSPEND {
794            return true;
795        }
796
797        if is_eof {
798            matches!(
799                self.opcode,
800                op::EXTCALL | op::EXTDELEGATECALL | op::EXTSTATICCALL | op::EOFCREATE
801            )
802        } else {
803            matches!(
804                self.opcode,
805                op::CALL
806                    | op::CALLCODE
807                    | op::DELEGATECALL
808                    | op::STATICCALL
809                    | op::CREATE
810                    | op::CREATE2
811            )
812        }
813    }
814}

816bitflags::bitflags! {
817    /// [`InstrData`] flags.
818    #[derive(Clone, Copy, Debug, Default, PartialEq, Eq)]
819    pub(crate) struct InstFlags: u8 {
820        /// The `JUMP`/`JUMPI` target is known at compile time.
821        /// This is implied for other jump instructions which are always static.
822        const STATIC_JUMP = 1 << 0;
823        /// The jump target is known to be invalid.
824        /// Always returns [`InstructionResult::InvalidJump`] at runtime.
825        const INVALID_JUMP = 1 << 1;
826        /// The instruction is a target of at least one `RJUMP*` instruction.
827        const EOF_JUMPDEST = 1 << 2;
828
829        /// The instruction is disabled in this EVM version.
830        /// Always returns [`InstructionResult::NotActivated`] at runtime.
831        const DISABLED = 1 << 3;
832        /// The instruction is unknown.
833        /// Always returns [`InstructionResult::NotFound`] at runtime.
834        const UNKNOWN = 1 << 4;
835        /// The instruction is only enabled in EOF bytecodes.
836        /// Always returns [`InstructionResult::EOFOpcodeDisabledInLegacy`] at runtime.
837        const EOF_ONLY = 1 << 5;
838
839        /// Skip generating instruction logic, but keep the gas calculation.
840        const SKIP_LOGIC = 1 << 6;
841        /// Don't generate any code.
842        const DEAD_CODE = 1 << 7;
843    }
844}

846fn bitvec_as_bytes<T: bitvec::store::BitStore, O: bitvec::order::BitOrder>(
847    bitvec: &BitVec<T, O>,
848) -> &[u8] {
849    slice_as_bytes(bitvec.as_raw_slice())
850}
851
852fn slice_as_bytes<T>(a: &[T]) -> &[u8] {
853    unsafe { std::slice::from_raw_parts(a.as_ptr().cast(), std::mem::size_of_val(a)) }
854}
855
856fn get_two_mut<T>(sl: &mut [T], idx_1: usize, idx_2: usize) -> (&mut T, &mut T) {
857    assert!(idx_1 != idx_2 && idx_1 < sl.len() && idx_2 < sl.len());
858    let ptr = sl.as_mut_ptr();
859    unsafe { (&mut *ptr.add(idx_1), &mut *ptr.add(idx_2)) }
860}
